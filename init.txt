using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;

namespace IPV4HeaderReader
{
    class Size
    {
        public int bytes
        {
            get;
            set;
        }
    }
    class Offset
    {
        public int bytes
        {
            get;
            set;
        }
    }

    class Type
    {
        public bool bigEndian
        {
            get;
            set;
        }
        public String type
        {
            get;
            set;
        }
    }
    
    class Content
    {
        public Content()
        {
            type = new Type();
        }
        public Type type
        {
            get;
            set;
        }

        public String data
        {
            get;
            set;
        }
    }
    
    class Structure
    {
        public Structure()
        {
            size=new Size();
            offset=new Offset();
            content = new Content();
        }

        public string name
        {
            get;
            set;
        }
        public Size size
        {
            get;
            set;
        }
        public Offset offset
        {
            get;
            set;
        }
        public Structure[] arrayOfStructures
        {
            get;
            set;
        }
        public Content content
        {
            get;
            set;
        }
    }
